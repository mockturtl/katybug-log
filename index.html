Katybug-log

XMLSocket connection to Powerflasher SOS using the Flex logging framework.

Download this project as a .zip file Download this project as a tar.gz file

Usage

Download SOSmax. It's Java.

Copy the source to your project's src directory, or compile a swc and drop it in lib.

package {  
  public class MyClass {
    import com.katybug.logging.LogUtil;
    import com.katybug.logging.SOSLogTarget;
    import mx.logging.ILogger;
    import mx.logging.Log;

    private static const LOG:ILogger = Log.getLogger(LogUtil.categoryFor(prototype));

    public function Main() {
      Log.addTarget(new SOSLogTarget());
    }

    public function demo() { 

      LOG.debug("Hello World!");
      LOG.info("Give any class a {0} instance", LOG);
      LOG.warn("Looks like trouble, unfold me for details...\n\tSee?");

      try {
        throw new Error("Something went wrong", 54321);
      } catch (e:Error) {
        LOG.error("{1}: {0}", e.message, e.errorID);
        LOG.fatal("Warp core breach.  You have been eaten by a grue.");
      }

    }  
  }
}
